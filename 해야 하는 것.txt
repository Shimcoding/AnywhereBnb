		function fn_idChk(){
			$.ajax({
				url : "/bnb/register/idChk",
				type : "post",
				dataType : "json",
				data : {"id" : $("#id").val()},
				success : function(data){

						alert("사용가능한 아이디입니다.");
					}
				}
			})
			
		}
